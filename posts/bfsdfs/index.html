# 너비 우선탐색
def bfs(s):
    # 양방향 큐인 덱을 생성해서 첫째 노드를 추가
    q= deque();q.append(s)
    # 첫째 노드 방문
    visit[s] = 1
    # 시작노드 처음 출력
    print(s, end=' ')
    # 댁에 요소가 있는경우
    while q:
        # 앞에서 빠진 원소가 현재 위치
        here = q.popleft()
        # 현재 노드 번째 배열에서
        for w in G[here]:
            # 현재노드에 있는 원소번째들이 방문을 하지 않은 경우
            if not visit[w]:
                # 방문으로 표시
                visit[w] = 1
                # 댁에 추가
                q.append(w)
                print(w, end=' ')

# 정점갯수, 간선개수, 탐색시작 정점 번호
